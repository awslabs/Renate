20class PromptPool(nn.Module): 

21 """Implements a pool of prompts to be used in for S-Prompts. 

22 

23 Args: 

24 prompt_size: Equivalent to number of input tokens used per update . Defaults to 10. 

25 embedding_size: Hidden size of the transformer used.. Defaults to 768. 

26 current_update_id: Current update it. Used to init number of prompts. Defaults to 0. 

27 """ 

28 

29 def __init__( 

30 self, prompt_size: int = 10, embedding_size: int = 768, current_update_id: int = 0 

31 ) -> None: 

32 super().__init__() 

33 self._M = prompt_size 

34 self._embedding_size = embedding_size 

35 self._curr_task = current_update_id 

36 

37 self._pool = nn.ParameterDict() 

38 for id in range(self._curr_task): 

39 # This always needs to be intialized as the torch's state dict only restores for an 

40 # existing Parameter. 

41 self._pool[f"{id}"] = nn.Parameter( 

42 torch.nn.init.kaiming_uniform_( 

43 torch.empty((self._M, self._embedding_size)), a=math.sqrt(5) 

44 ) 

45 ) 

46 

47 self._pool.requires_grad_(True) 

48 

49 def forward(self, id: int) -> torch.nn.Parameter: 

50 return self._pool[f"{id}"] 

51 

52 def get_params(self, id: int) -> List[torch.nn.Parameter]: 

53 return [self._pool[f"{id}"]] 

54 

55 def increment_task(self) -> None: 

56 self._pool[f"{len(self._pool)}"] = nn.Parameter( 

57 torch.empty((self._M, self._embedding_size)).uniform_(-1, 1) 

58 ) 

59 self._pool.requires_grad_(True)

62class SPromptTransformer(RenateBenchmarkingModule): 

63 """Implements Transformer Model for S-Prompts as described in Wang, Yabin, et.al ."S-prompts 

64 learning with pre-trained transformers: An occam’s razor for domain incremental learning." 

65 Advances in Neural Information Processing Systems 35 (2022): 5682-5695. 

66 

67 Args: 

68 pretrained_model_name_or_path: A string that denotes which pretrained model from the HF hub 

69 to use. 

70 image_size: Image size. Used if `pretrained_model_name_or_path` is not set . 

71 patch_size: Patch size to be extracted. Used if `pretrained_model_name_or_path` is not set . 

72 num_layers: Num of transformer layers. Used only if `pretrained_model_name_or_path` is not 

73 set . 

74 num_heads: Num heads in MHSA. Used only if `pretrained_model_name_or_path` is not set . 

75 hidden_dim: Hidden dimension of transformers. Used only if `pretrained_model_name_or_path` 

76 is not set . 

77 mlp_dim: _description_. Used only if `pretrained_model_name_or_path` is not set . 

78 dropout: _description_. Used only if `pretrained_model_name_or_path` is not set . 

79 attention_dropout: _description_. Used only if `pretrained_model_name_or_path` is not set . 

80 num_outputs: Number of output classes of the output. Defaults to 10. 

81 prediction_strategy: Continual learning strategies may alter the prediction at train or test 

82 time. Defaults to None. 

83 add_icarl_class_means: If ``True``, additional parameters used only by the 

84 ``ICaRLModelUpdater`` are added. Only required when using that updater. 

85 prompt_size: Equivalent to number of input tokens used per update . Defaults to 10. 

86 task_id: Internal variable used to increment update id. Shouldn't be set by user. 

87 Defaults to 0. 

88 clusters_per_task: Number clusters in k-means used for task identification. Defaults to 5. 

89 per_task_classifier: Flag to share or use a common classifier head for all tasks. 

90 Defaults to False. 

91 """ 

92

94 self,

97 patch_size: int = 4, 

98 num_layers: int = 12, 

99 num_heads: int = 12, 

100 hidden_dim: int = 768, 

101 mlp_dim: int = 3072, 

102 dropout: float = 0.1, 

103 attention_dropout: float = 0.1, 

104 num_outputs: int = 10, 

105 prediction_strategy: Optional[PredictionStrategy] = None, 

106 add_icarl_class_means: bool = True, 

107 prompt_size: int = 10,

110 per_task_classifier: bool = False, 

111 ):

119 mlp_dim=mlp_dim,

121 attention_dropout=attention_dropout, 

122 num_outputs=num_outputs, 

123 prediction_strategy=prediction_strategy, 

124 add_icarl_class_means=add_icarl_class_means, 

125 ) 

126 super().__init__( 

127 embedding_size=transformer.transformer._embedding_size, 

128 num_outputs=num_outputs, 

129 constructor_arguments=dict( 

130 **transformer.transformer._constructor_arguments, 

131 prompt_size=prompt_size,

153 embedding_size=self._embedding_size, 

154 ) 

155 self._backbone["transformer"].requires_grad_(False) 

156 self._backbone["prompt_pool"].requires_grad_(True) 

157 

158 self._backbone["classifier"] = SharedMultipleLinear( 

159 self._embedding_size, 

160 self._num_outputs, 

161 share_parameters=not self._per_task_classifier, 

162 num_updates=self._task_id + 1, 

163 ) 

164 

165 self._tasks_params = nn.ModuleDict( 

166 {k: nn.Identity() for k, _ in self._tasks_params.items()} 

167 ) 

168 

169 self._backbone.forward = self._forward_for_monkey_patching 

170 self.task_ids = None 

171 

172 def increment_task(self) -> None: 

173 # This cannot be a part of add_task_params as the super.__init__ function calls 

174 # add_task_params, and thus we would be trying parameters to the non-existent 

175 # self.s_prompts 

176 self._backbone["prompt_pool"].increment_task()

178 def _forward_for_monkey_patching( 

179 self, x: Union[torch.Tensor, Dict[str, Any]], task_id: str = None 

180 ) -> torch.Tensor: 

181 prompt = None 

182 task_ids = None 

183 if self.training: 

184 prompt = self._backbone["prompt_pool"](self._task_id) 

185 else: 

186 task_ids = self._task_id_method.infer_task(self._backbone["transformer"](x)) 

187 if task_ids is not None: 

188 prompt = torch.stack([self._backbone["prompt_pool"](i) for i in task_ids]) 

189 self.task_ids = task_ids.detach().cpu().numpy() 

190 

191 features = self._backbone["transformer"](x, prompt) 

192 

193 # additional logic for separate classifiers 

194 # a. This forward returns logits directly, and the RenateBenchmarkingModule's _task_params 

195 # now are identities. Thus, the overall operation is still the network forward pass. 

196 # b. Additional handling of params is not needed as backbone's params will return all the 

197 # necessary elements. 

198 

199 if self.training: 

200 logits = self._backbone["classifier"][f"{self._task_id}"](features) 

201 elif task_ids is not None: 

202 logits = torch.cat( 

203 [ 

204 self._backbone["classifier"][f"{t}"](feat.unsqueeze(0)) 

205 for t, feat in zip(task_ids, features) 

206 ] 

207 ) 

208 else: 

209 logits = self._backbone["classifier"]["0"](features) 

210 

211 return logits

219 self._constructor_arguments["task_id"] = self._task_id + 1

221 def features(self, x: torch.Tensor) -> torch.Tensor: 

222 return self._backbone["transformer"](x)
