3import torch 

4from scipy.stats import kstest 

5 

6from renate.shift.detector import ShiftDetectorWithFeatureExtractor 

7 

8 

9class KolmogorovSmirnovCovariateShiftDetector(ShiftDetectorWithFeatureExtractor): 

10 """A Kolmogorov-Smirnov (KS) test on each feature. 

11 

12 A KS test is a univariate two-sample test, which we perform separately for each feature. To 

13 aggregate these tests without running into multiple-testing problems, we use a Bonferroni 

14 correction, as proposed in 

15 

16 Rabanser et al. Failing Loudly: An Empirical Study of Methods for Detecting Dataset Shift. 

17 NeurIPS 2019. 

18 """ 

19 

20 def _fit_with_features(self, X: torch.Tensor) -> None: 

21 self._X_ref = X 

22 

23 def _score_with_features(self, X: torch.Tensor) -> float: 

24 n_features = X.size(1) 

25 p_vals = [ 

26 kstest(X[:, i].numpy(), self._X_ref[:, i].numpy()).pvalue for i in range(n_features) 

27 ] 

28 # Bonferroni correction: Reject only if the minimal p-value among the multiple tests is 

29 # lower than `alpha / num_tests`, where `alpha` is the significance level. Equivalently, we 

30 # multiply the p-value by `num_tests`. 

31 p_val = min(1.0, min(p_vals) * n_features) 

32 return 1.0 - p_val
