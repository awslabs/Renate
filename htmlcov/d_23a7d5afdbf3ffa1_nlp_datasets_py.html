15class TorchTextDataModule(RenateDataModule): 

16 """Dataset with data from torchtext. 

17 

18 Source: https://pytorch.org/text/stable/datasets.html 

19 

20 Args: 

21 data_path: the path to the folder containing the dataset files. 

22 src_bucket: the name of the s3 bucket. If not provided, downloads the data from original 

23 source. 

24 src_object_name: the folder path in the s3 bucket. 

25 dataset_name: Name of the torchvision dataset. 

26 val_size: Fraction of the training data to be used for validation. 

27 seed: Seed used to fix random number generation. 

28 """ 

29 

30 dataset_dict = { 

31 "AG_NEWS": (torchtext.datasets.AG_NEWS, "ag_news_csv"), 

32 "AmazonReviewFull": (torchtext.datasets.AmazonReviewFull, "amazon_review_full_csv"), 

33 "DBpedia": (torchtext.datasets.DBpedia, "dbpedia_csv"), 

34 } 

35 

36 def __init__( 

37 self, 

38 data_path: Union[Path, str], 

39 src_bucket: Optional[str] = None, 

40 src_object_name: Optional[str] = None, 

41 dataset_name: str = "AG_NEWS", 

42 val_size: float = defaults.VALIDATION_SIZE, 

43 seed: int = defaults.SEED, 

44 ): 

45 super(TorchTextDataModule, self).__init__( 

46 data_path, 

47 src_bucket=src_bucket, 

48 src_object_name=src_object_name, 

49 val_size=val_size, 

50 seed=seed, 

51 ) 

52 if dataset_name not in TorchTextDataModule.dataset_dict: 

53 raise ValueError(f"Dataset {self._dataset_name} currently not supported.") 

54 self._dataset_name = dataset_name 

55 

56 def prepare_data(self) -> None: 

57 """Download data. 

58 

59 If s3 bucket is given, the data is downloaded from s3, otherwise from the original source. 

60 """ 

61 cls, dataset_pathname = TorchTextDataModule.dataset_dict[self._dataset_name] 

62 if self._src_bucket is not None: 

63 download_folder_from_s3( 

64 src_bucket=self._src_bucket, 

65 src_object_name=self._src_object_name, 

66 dst_dir=os.path.join(self._data_path, dataset_pathname), 

67 ) 

68 else: 

69 # Use torchtext to download. 

70 cls(root=self._data_path, split="train") 

71 cls(root=self._data_path, split="test") 

72 

73 def setup(self) -> None: 

74 """Set up train, test and val datasets.""" 

75 cls, _ = TorchTextDataModule.dataset_dict[self._dataset_name] 

76 train_data = to_map_style_dataset(cls(root=self._data_path, split="train")) 

77 self._train_data, self._val_data = self._split_train_val_data(train_data) 

78 self._test_data = to_map_style_dataset(cls(root=self._data_path, split="test"))
