15class RenateBenchmarkingModule(RenateModule, ABC): 

16 """Base class for all models provided by Renate. 

17 

18 This class ensures that each models works with all ModelUpdaters when using the benchmarking 

19 feature of Renate. New models can extend this class or alternatively extend the RenateModule 

20 and make sure they are compatible with the considered ModelUpdater. 

21 

22 Args: 

23 embedding_size: Representation size of the model after the backbone. 

24 num_outputs: The number of outputs of the model. 

25 constructor_arguments: Arguments needed to instantiate the model. 

26 loss_fn: The loss function to be optimized during the training. 

27 prediction_strategy: By default a forward pass through the model. Some ModelUpdater must 

28 be combined with specific prediction strategies to work as intended. 

29 add_icarl_class_means: Specific parameters for iCaRL. Can be set to ``False`` if any other 

30 ModelUpdater is used. 

31 """ 

32 

33 def __init__( 

34 self, 

35 embedding_size: int, 

36 num_outputs: int, 

37 constructor_arguments: dict, 

38 loss_fn: torch.nn.Module, 

39 prediction_strategy: Optional[PredictionStrategy] = None, 

40 add_icarl_class_means: bool = True, 

41 ): 

42 constructor_arguments["num_outputs"] = num_outputs 

43 constructor_arguments["add_icarl_class_means"] = add_icarl_class_means 

44 super().__init__( 

45 constructor_arguments=constructor_arguments, 

46 loss_fn=loss_fn, 

47 ) 

48 self._embedding_size = embedding_size 

49 self._num_outputs = num_outputs 

50 self._prediction_strategy = prediction_strategy 

51 self._tasks_params: torch.nn.ModuleDict = torch.nn.ModuleDict() 

52 self.add_task_params(defaults.TASK_ID) 

53 if add_icarl_class_means: 

54 self.class_means = torch.nn.Parameter( 

55 torch.zeros((embedding_size, num_outputs)), requires_grad=False 

56 ) 

57 

58 def forward(self, x: torch.Tensor, task_id: str = defaults.TASK_ID) -> torch.Tensor: 

59 x = self.get_backbone(task_id=task_id)(x) 

60 if isinstance(self._prediction_strategy, ICaRLClassificationStrategy): 

61 return self._prediction_strategy(x, self.training, class_means=self.class_means) 

62 else: 

63 assert ( 

64 self._prediction_strategy is None 

65 ), f"Unknown prediction strategy of type {type(self._prediction_strategy)}." 

66 return self.get_predictor(task_id)(x) 

67 

68 def get_backbone(self, task_id: str = defaults.TASK_ID) -> nn.Module: 

69 """Returns the model without the prediction head.""" 

70 return self._backbone 

71 

72 def get_predictor(self, task_id: str = defaults.TASK_ID) -> nn.Module: 

73 """Returns the model without the backbone.""" 

74 return self._tasks_params[task_id] 

75 

76 def _add_task_params(self, task_id: str = defaults.TASK_ID) -> None: 

77 """Adds new parameters associated to a specific task to the model.""" 

78 self._tasks_params[task_id] = torch.nn.Linear(self._embedding_size, self._num_outputs) 

79 

80 def get_params(self, task_id: str = defaults.TASK_ID) -> List[torch.nn.Parameter]: 

81 """Returns the list of parameters for the core model and a specific `task_id`.""" 

82 return list(self.get_backbone(task_id=task_id).parameters()) + list( 

83 self.get_predictor(task_id=task_id).parameters() 

84 ) 

85 

86 def get_extra_state(self) -> Any: 

87 """Get the constructor_arguments, loss and task ids necessary to reconstruct the model.""" 

88 extra_state = super().get_extra_state() 

89 extra_state["prediction_strategy"] = self._prediction_strategy 

90 return extra_state 

91 

92 def set_extra_state(self, state: Any): 

93 """Extract the content of the ``_extra_state`` and set the related values in the module.""" 

94 super().set_extra_state(state) 

95 self._prediction_strategy = state["prediction_strategy"]
