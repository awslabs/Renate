17class TuningLoggerCallback(TunerCallback): 

18 """Syne Tune Logging Callback when running hyperparameter optimization. 

19 

20 Will report whenever improved results are obtained. If no better configuration is received 

21 within the last ten minutes, it will send a message to indicate it is still running. 

22 """ 

23 

24 def __init__(self, mode: str, metric: str): 

25 self._mode = mode 

26 self._metric = metric 

27 self._best_score = float("inf") * (1 if self._mode == "min" else -1) 

28 self._last_log = time.time() 

29 

30 def _log(self, message: str) -> None: 

31 """Helper function to log a message.""" 

32 logger.info(message) 

33 self._last_log = time.time() 

34 

35 def on_trial_result(self, trial: Trial, status: str, result: Dict, decision: str) -> None: 

36 """Called whenever the tuner receives a new (intermediate) observation.""" 

37 if self._mode == "min": 

38 is_better_config = result[self._metric] < self._best_score 

39 self._best_score = min(result[self._metric], self._best_score) 

40 else: 

41 is_better_config = result[self._metric] > self._best_score 

42 self._best_score = max(result[self._metric], self._best_score) 

43 if is_better_config: 

44 self._log(f"Metric `{self._metric}` improved to {self._best_score}.") 

45 if self._last_log < time.time() - 600: 

46 self._log("No better configuration found since last update. Continue searching...")

62def redirect_to_tmp(uri: str) -> str: 

63 """Changes uri in /opt/ml to /tmp. 

64 

65 Syne Tune stores checkpoints by default in /opt/ml when running on SageMaker. While we want to 

66 store checkpoints, we have no interest in uploading them to S3. Therefore, this function changes 

67 the location to /tmp instead. 

68 """ 

69 if "SM_MODEL_DIR" in os.environ: # If running on sagemaker, redirect checkpoints to /tmp 

70 assert uri.startswith("/opt/ml") 

71 uri = "/tmp" + uri[7:] 

72 return uri
