3from dataclasses import dataclass 

4from typing import Any, Dict, List, Tuple, Union 

5 

6import torch 

7from torch.utils.data import default_collate 

8from transformers import BatchEncoding, DataCollatorWithPadding 

9 

10 

11@dataclass 

12class DataCollatorWithPaddingForWildTime(DataCollatorWithPadding): 

13 """A data collator class that can handle wild time data (non-standard) batches. 

14 

15 This adds to the `transformer` library's DataCollatorWithPadding. That data collator expects 

16 data in a standard HF format. Wild time data format is slightly different: We get a 

17 tuple of BatchEncoding (dict) and a class label. When being read from a buffer, an additional 

18 metadata attribute is present. These cases are not handled by the orig data collator. 

19 The code here only separates the input data into format original collator can handle and 

20 undoes the data packing: see parts after super().__call__. 

21 """ 

22 

23 def __call__(self, features: List[Dict[str, Any]]) -> Dict[str, Any]: 

24 # first_element type determines if data sources 

25 # known types : {features}, class (from wildtimedataset) 

26 # : ({features}, class), metadata (from a replay buffer : val data?) 

27 # : index, ({features}, class) (by _EnumeratedDataset in BaseER traindata) 

28 # Note that {features} can be a dict or BatchEncoding. 

29 first_element = features[0][0] 

30 assert len(features[0]) == 2, "Input to collator should contain only two elements" 

31 if isinstance(first_element, (BatchEncoding, dict)): 

32 # This is normal dataset and not a buffer with metadata, do the default things 

33 return self._collate_hf_class_tuples(features) 

34 elif isinstance(first_element, tuple): 

35 # this has metadata possibly. 

36 collated, labels = self._collate_hf_class_tuples([elem[0] for elem in features]) 

37 metadata_to_collate = [elem[1] for elem in features] 

38 collated_metadata = default_collate(metadata_to_collate) 

39 return (collated, labels), collated_metadata 

40 elif isinstance(first_element, (torch.Tensor, int)): 

41 ## this is EnumeratedDataset. 

42 collated_indices = default_collate([elem[0] for elem in features]) 

43 collated, labels = self._collate_hf_class_tuples([elem[1] for elem in features]) 

44 return collated_indices, (collated, labels) 

45 

46 else: 

47 raise ValueError( 

48 f"Unknown structure to collate. Got {first_element} of {type(first_element)}" 

49 ) 

50 

51 def _collate_hf_class_tuples( 

52 self, features: Tuple[Union[BatchEncoding, Dict[str, Any]], Union[torch.Tensor, int]] 

53 ): 

54 to_be_collated = [] 

55 for elem in features: 

56 elem[0]["labels"] = elem[1] 

57 to_be_collated.append(elem[0]) 

58 collated = super().__call__(to_be_collated) 

59 labels = collated.pop("labels") 

60 return collated, labels
