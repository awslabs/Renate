262def _get_transfer_learning_task_evaluations( 

263 tuning_results: pd.DataFrame, 

264 config_space: Dict[str, Any], 

265 metric: str, 

266 max_epochs: int, 

267) -> Optional[TransferLearningTaskEvaluations]: 

268 """Converts data frame with training results of a single update step into 

269 `TransferLearningTaskEvaluations`. 

270 

271 Args: 

272 tuning_results: Results of previous hyperparameter optimization runs. 

273 config_space: Configuration space used. 

274 metric: The metric to be optimized. This will be the only metric added to 

275 `TransferLearningTaskEvaluations`. 

276 max_epochs: Length of the learning curve. Learning curves will be padded with `np.nan` 

277 to this length. 

278 Returns: 

279 `TransferLearningTaskEvaluations` contains a `pd.DataFrame` of hyperparameters, a `np.array` 

280 of shape `(num_hyperparameters, 1, max_epochs, 1)` which contains the learning curves for 

281 `metric` for each hyperparameter. 

282 """ 

283 for config in RENATE_CONFIG_COLUMNS: 

284 config_with_prefix = f"config_{config}" 

285 if config in config_space: 

286 tuning_results[config_with_prefix] = config_space[config] 

287 elif config_with_prefix in tuning_results: 

288 del tuning_results[config_with_prefix] 

289 if ( 

290 metric not in tuning_results 

291 or not len(tuning_results) 

292 or not set([f"config_{config_name}" for config_name in config_space]).issubset( 

293 tuning_results 

294 ) 

295 ): 

296 return None 

297 

298 def pad_same_length(x): 

299 if len(x) == max_epochs: 

300 return x 

301 return x + (max_epochs - len(x)) * [np.nan] 

302 

303 hyperparameter_names = [ 

304 hyperparameter for hyperparameter in tuning_results if hyperparameter.startswith("config_") 

305 ] 

306 hyperparameters = tuning_results[hyperparameter_names] 

307 hyperparameters.columns = [ 

308 hyperparameter_name[7:] for hyperparameter_name in hyperparameters.columns 

309 ] 

310 hyperparameters = hyperparameters[list(config_space)] 

311 hyperparameter_rows_with_nan = hyperparameters.isna().any(axis=1) 

312 hyperparameter_duplicates = hyperparameters.duplicated() 

313 objectives_evaluations = np.array(tuning_results[metric].apply(pad_same_length).to_list()) 

314 objectives_evaluations = np.expand_dims(objectives_evaluations, axis=(1, 3)) 

315 objectives_evaluations_with_nan_only = np.all(np.isnan(objectives_evaluations), axis=2).reshape( 

316 -1 

317 ) 

318 valid_hyperparameters = np.logical_and( 

319 np.logical_and(~hyperparameter_rows_with_nan, ~objectives_evaluations_with_nan_only), 

320 ~hyperparameter_duplicates, 

321 ) 

322 if not np.any(valid_hyperparameters): 

323 return None 

324 hyperparameters = hyperparameters[valid_hyperparameters] 

325 objectives_evaluations = objectives_evaluations[valid_hyperparameters] 

326 return TransferLearningTaskEvaluations( 

327 configuration_space=config_space, 

328 hyperparameters=hyperparameters, 

329 objectives_names=[metric], 

330 objectives_evaluations=objectives_evaluations, 

331 )

334def _load_tuning_history( 

335 input_state_url: str, config_space: Dict[str, Any], metric: str 

336) -> Dict[str, TransferLearningTaskEvaluations]: 

337 """Loads the tuning history in a list where each entry of the list is the tuning history of one 

338 update. 

339 

340 Args: 

341 input_state_url: Location of state. Will check at this location of a tuning history exists. 

342 config_space: The configuration space defines which parts of the tuning history to load. 

343 metric: Only the defined metric of the tuning history will be loaded. 

344 Returns: 

345 Returns an empty list if no previous tuning history exists or it does not match the current 

346 `config_space`. The list contains an instance of `TransferLearningTaskEvaluations` for each 

347 update that contains matching data. 

348 """ 

349 if input_state_url is None or not Path(defaults.hpo_file(input_state_url)).exists(): 

350 return {} 

351 tuning_results = pd.read_csv(defaults.hpo_file(input_state_url)) 

352 hyperparameter_names = [ 

353 hyperparameter for hyperparameter in tuning_results if hyperparameter.startswith("config_") 

354 ] 

355 agg_dict = { 

356 hyperparameter_name: lambda x: x.iloc[0] for hyperparameter_name in hyperparameter_names 

357 } 

358 agg_dict[metric] = list 

359 max_epochs = tuning_results["epoch"].max() 

360 task_evaluations = {} 

361 for update_id in tuning_results["update_id"].unique(): 

362 task_evaluation = _get_transfer_learning_task_evaluations( 

363 tuning_results=tuning_results[tuning_results["update_id"] == update_id] 

364 .groupby(["trial_id"]) 

365 .agg(agg_dict), 

366 config_space=config_space, 

367 metric=metric, 

368 max_epochs=max_epochs, 

369 ) 

370 if task_evaluation is not None: 

371 task_evaluations[f"update_id_{update_id}"] = task_evaluation 

372 return task_evaluations

375def _merge_tuning_history( 

376 new_tuning_results: pd.DataFrame, old_tuning_results: Optional[pd.DataFrame] = None 

377) -> pd.DataFrame: 

378 """Merges old tuning history with training results from current chunk. 

379 

380 `update_id` identifies the update step in the csv file. This allows creating the metadata 

381 required for transfer hyperparameter optimization. 

382 """ 

383 update_id = 0 if old_tuning_results is None else old_tuning_results["update_id"].max() + 1 

384 new_tuning_results.insert(0, "update_id", update_id) 

385 if old_tuning_results is not None: 

386 return pd.concat([old_tuning_results, new_tuning_results], axis=0, ignore_index=True) 

387 return new_tuning_results

390def _teardown_tuning_job( 

391 backend: LocalBackend, 

392 config_space: Dict[str, Union[Domain, int, float, str]], 

393 job_name: str, 

394 input_state_url: Optional[str] = None, 

395 output_state_url: Optional[str] = None, 

396) -> None: 

397 """Update lifelong hyperparameter optimization results, save state and clean up disk.""" 

398 experiment_folder = redirect_to_tmp(str(experiment_path(job_name))) 

399 if output_state_url is not None: 

400 experiment = load_experiment(job_name) 

401 try: 

402 best_trial_id = experiment.best_config()["trial_id"] 

403 if is_syne_tune_config_space(config_space): 

404 logger.info( 

405 "Best hyperparameter settings: " 

406 f"{best_hyperparameters(experiment, config_space)}" 

407 ) 

408 except AttributeError: 

409 raise RuntimeError( 

410 "Not a single training run finished. This may have two reasons:\n" 

411 "1) The provided tuning time is too short.\n" 

412 "2) There is a bug in the training script." 

413 + "\n\nLogs (stdout):\n\n{}".format("".join(backend.stdout(0))) 

414 + "\n\nLogs (stderr):\n\n{}".format("".join(backend.stderr(0))) 

415 ) 

416 output_state_folder = defaults.output_state_folder( 

417 f"{experiment_folder}/{best_trial_id}/checkpoints" 

418 ) 

419 old_tuning_results = ( 

420 pd.read_csv(defaults.hpo_file(input_state_url)) 

421 if input_state_url is not None and os.path.exists(defaults.hpo_file(input_state_url)) 

422 else None 

423 ) 

424 tuning_results = _merge_tuning_history(experiment.results, old_tuning_results) 

425 tuning_results.to_csv(defaults.hpo_file(output_state_folder), index=False) 

426 move_to_uri(output_state_folder, output_state_url) 

427 logger.info(f"Renate state is available at {output_state_url}.") 

428 shutil.rmtree(experiment_folder, ignore_errors=True) 

429 shutil.rmtree(experiment_path(job_name), ignore_errors=True)

432def _verify_validation_set_for_hpo_and_checkpointing( 

433 config_space: Dict[str, Any], 

434 config_file: str, 

435 tune_hyperparameters: bool, 

436 metric: str, 

437 mode: defaults.SUPPORTED_TUNING_MODE_TYPE, 

438 working_directory: str, 

439) -> Tuple[str, defaults.SUPPORTED_TUNING_MODE_TYPE]: 

440 """Checks if validation set is provided when needed and updates config_space such that 

441 checkpointing works. 

442 

443 If a validation set exists, the metric is forwarded such that we store the checkpoint which 

444 performs best on validation. This is a side effect changing `config_space`. Otherwise, the 

445 checkpoint of the last epoch is used. 

446 

447 Returns: 

448 Metric and mode used by the Syne Tune tuner. If there is no validation set, returns 

449 `("train_loss", "min")`. 

450 Raises: 

451 AssertionError: If `tune_hyperparameters` is True but no validation set is provided. 

452 """ 

453 config_module = import_module("config_module", config_file) 

454 data_module = get_and_prepare_data_module( 

455 config_module, 

456 data_path=defaults.data_folder(working_directory), 

457 **get_data_module_fn_kwargs(config_module, config_space, cast_arguments=True), 

458 ) 

459 data_module.setup() 

460 val_exists = data_module.val_data() is not None 

461 assert ( 

462 val_exists or not tune_hyperparameters 

463 ), "Provide a validation set to optimize hyperparameters." 

464 if val_exists: 

465 config_space["metric"] = metric 

466 config_space["mode"] = mode 

467 return metric, mode 

468 elif not val_exists and metric is not None: 

469 warnings.warn("No need to pass `metric`. Without a validation set, it is not used.") 

470 return "train_loss", "min"

473def _create_scheduler( 

474 scheduler: Union[str, Type[TrialScheduler]], 

475 config_space: Dict[str, Any], 

476 metric: str, 

477 mode: defaults.SUPPORTED_TUNING_MODE_TYPE, 

478 seed: int, 

479 scheduler_kwargs: Optional[Dict[str, Any]] = None, 

480 input_state_url: Optional[str] = None, 

481) -> TrialScheduler: 

482 scheduler_kwargs = scheduler_kwargs or {} 

483 if isinstance(scheduler, str): 

484 hyperband_scheduler_kwargs = {"max_resource_attr": "max_epochs", "resource_attr": "epoch"} 

485 scheduler_classes = { 

486 "asha": {"scheduler": ASHA, "scheduler_kwargs": hyperband_scheduler_kwargs}, 

487 "bo": {"scheduler": FIFOScheduler, "scheduler_kwargs": {"searcher": "bayesopt"}}, 

488 "rush": {"scheduler": RUSHScheduler, "scheduler_kwargs": hyperband_scheduler_kwargs}, 

489 "random": {"scheduler": FIFOScheduler, "scheduler_kwargs": {"searcher": "random"}}, 

490 } 

491 assert ( 

492 scheduler in scheduler_classes 

493 ), f"Unknown scheduler {scheduler}. Options: {list(scheduler_classes)}." 

494 scheduler_kwargs.update(scheduler_classes[scheduler]["scheduler_kwargs"]) 

495 scheduler = scheduler_classes[scheduler]["scheduler"] 

496 if "transfer_learning_evaluations" in inspect.getfullargspec(scheduler.__init__).args: 

497 scheduler_kwargs["transfer_learning_evaluations"] = _load_tuning_history( 

498 input_state_url=input_state_url, config_space=config_space, metric=metric 

499 ) 

500 if scheduler_kwargs["transfer_learning_evaluations"]: 

501 logger.info( 

502 f"Using information of {len(scheduler_kwargs['transfer_learning_evaluations'])} " 

503 "previous tuning jobs to accelerate this job." 

504 ) 

505 return scheduler( 

506 config_space=config_space, 

507 mode=mode, 

508 metric=metric, 

509 random_seed=seed, 

510 **scheduler_kwargs, 

511 )

551 config_space["prepare_data"] = False 

552 if chunk_id is not None: 

553 config_space["chunk_id"] = chunk_id 

554 config_space["task_id"] = task_id 

555 config_space["working_directory"] = working_directory 

556 config_space["seed"] = seed 

557 config_space["accelerator"] = accelerator 

558 config_space["devices"] = devices 

559 config_space["strategy"] = strategy 

560 config_space["precision"] = precision 

561 config_space["deterministic_trainer"] = deterministic_trainer 

562 config_space["gradient_clip_val"] = gradient_clip_val 

563 config_space["gradient_clip_algorithm"] = gradient_clip_algorithm 

564 if input_state_url is not None: 

565 config_space["input_state_url"] = input_state_url 

566 

567 metric, mode = _verify_validation_set_for_hpo_and_checkpointing( 

568 config_space=config_space, 

569 config_file=config_file, 

570 tune_hyperparameters=tune_hyperparameters, 

571 metric=metric, 

572 mode=mode, 

573 working_directory=working_directory, 

574 ) 

575 

576 training_script = str(Path(renate.__path__[0]) / "cli" / "run_training.py") 

577 assert Path(training_script).is_file(), f"Could not find training script {training_script}." 

578 logger.info("Start updating the model.") 

579 if tune_hyperparameters: 

580 logger.info( 

581 f"Tuning hyperparameters with respect to {metric} ({mode}) for {max_time} seconds on " 

582 f"{n_workers} worker(s)." 

583 ) 

584 

585 backend = LocalBackend(entry_point=training_script, num_gpus_per_trial=devices) 

586 if scheduler is None or not tune_hyperparameters: 

587 if scheduler is not None: 

588 warnings.warn( 

589 "Configuration space contains exactly one configuration, custom scheduler is " 

590 "ignored." 

591 ) 

592 scheduler = defaults.scheduler(config_space=config_space, mode=mode, metric=metric) 

593 else: 

594 scheduler = _create_scheduler( 

595 scheduler=scheduler, 

596 config_space=config_space, 

597 metric=metric, 

598 mode=mode, 

599 seed=seed, 

600 scheduler_kwargs=scheduler_kwargs, 

601 input_state_url=input_state_url, 

602 ) 

603 logging_callback = ( 

604 TuningLoggerCallback(mode=mode, metric=metric) 

605 if tune_hyperparameters 

606 else TrainingLoggerCallback() 

607 ) 

608 tuner = Tuner( 

609 trial_backend=backend, 

610 scheduler=scheduler, 

611 stop_criterion=StoppingCriterion( 

612 max_wallclock_time=max_time, 

613 max_num_trials_started=max_num_trials_started, 

614 max_num_trials_completed=max_num_trials_completed, 

615 max_num_trials_finished=max_num_trials_finished, 

616 max_cost=max_cost, 

617 ), 

618 n_workers=n_workers, 

619 callbacks=[StoreResultsCallback(), logging_callback], 

620 ) 

621 

622 try: 

623 tuner.run() 

624 except ValueError: 

625 raise RuntimeError( 

626 "Tuning failed." 

627 + "\n\nLogs (stdout):\n\n{}".format("".join(backend.stdout(0))) 

628 + "\n\nLogs (stderr):\n\n{}".format("".join(backend.stderr(0))) 

629 ) 

630 

631 logger.info("All training is completed. Saving state...") 

632 

633 _teardown_tuning_job( 

634 backend=backend, 

635 config_space=config_space, 

636 job_name=tuner.name, 

637 input_state_url=input_state_url, 

638 output_state_url=output_state_url, 

639 ) 

640 

641 logger.info("Renate update completed successfully.") 

642 

643 return tuner
