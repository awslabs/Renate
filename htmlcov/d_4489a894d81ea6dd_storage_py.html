36class MemoryMappedTensorStorage(Storage): 

37 """A class implementing permanent storage of nested tensor datasets. 

38 

39 This implements storage for `length` data points consisting of nested tensors of fixed types 

40 and shapes. `Storage` implements `__len__` and `__getitem__` and therefore can be used as a 

41 torch `Datasets`. To populate the storage, it also implements `__setitem__`. It does _not_ keep 

42 track which slots have or have not been populated. 

43 

44 `Storage` is given a path to a directory, where it creates (or accesses, if they already exist) 

45 memory-mapped tensor files. 

46 

47 Args: 

48 directory: Path to a directory. 

49 data_point: Prototypical datapoint from which to infer shapes/dtypes. 

50 length: Number of items to be stored. 

51 """ 

52 

53 def __init__(self, directory: str, data_point: NestedTensors, length: int) -> None: 

54 super().__init__(directory, data_point, length) 

55 self._storage = self._create_mmap_tensors(directory, data_point, length) 

56 

57 @staticmethod 

58 def _create_mmap_tensors(path: str, data_point: NestedTensors, length: int) -> NestedTensors: 

59 if isinstance(data_point, torch.Tensor): 

60 os.makedirs(os.path.dirname(path), exist_ok=True) 

61 filename = f"{path}.pt" 

62 return mmap_tensor(filename, size=(length, *data_point.size()), dtype=data_point.dtype) 

63 elif isinstance(data_point, tuple): 

64 return tuple( 

65 MemoryMappedTensorStorage._create_mmap_tensors( 

66 os.path.join(path, f"{i}"), data_point[i], length 

67 ) 

68 for i in range(len(data_point)) 

69 ) 

70 elif isinstance(data_point, dict): 

71 return { 

72 key: MemoryMappedTensorStorage._create_mmap_tensors( 

73 os.path.join(path, key), data_point[key], length 

74 ) 

75 for key in data_point 

76 } 

77 else: 

78 raise TypeError(f"Expected nested tuple/dict of tensors, found {type(data_point)}.") 

79 

80 @staticmethod 

81 def _get(storage: NestedTensors, idx: int) -> NestedTensors: 

82 if isinstance(storage, torch.Tensor): 

83 return storage[idx] 

84 elif isinstance(storage, tuple): 

85 return tuple(MemoryMappedTensorStorage._get(t, idx) for t in storage) 

86 elif isinstance(storage, dict): 

87 return {key: MemoryMappedTensorStorage._get(t, idx) for key, t in storage.items()} 

88 else: 

89 raise TypeError(f"Expected nested tuple/dict of tensors, found {type(storage)}.") 

90 

91 def __getitem__(self, idx: int) -> NestedTensors: 

92 """Read the item stored at index `idx`.""" 

93 return self._get(self._storage, idx)

95 @staticmethod 

96 def _set(storage: NestedTensors, idx: int, data_point: NestedTensors) -> None: 

97 if isinstance(storage, torch.Tensor): 

98 assert isinstance(data_point, torch.Tensor) 

99 assert data_point.dtype is storage.dtype 

100 storage[idx] = data_point 

101 elif isinstance(storage, tuple): 

102 assert isinstance(data_point, tuple) 

103 assert len(data_point) == len(storage) 

104 for i in range(len(storage)): 

105 MemoryMappedTensorStorage._set(storage[i], idx, data_point[i]) 

106 elif isinstance(storage, dict): 

107 assert isinstance(data_point, dict) 

108 assert set(data_point.keys()) == set(storage.keys()) 

109 for key in storage: 

110 MemoryMappedTensorStorage._set(storage[key], idx, data_point[key]) 

111 else: 

112 raise TypeError(f"Expected nested tuple/dict of tensors, found {type(storage)}.")
