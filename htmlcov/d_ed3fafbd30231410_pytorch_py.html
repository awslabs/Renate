12def reinitialize_model_parameters(model: torch.nn.Module) -> None: 

13 """Reinitializes the parameters of a model. 

14 

15 This relies on all submodules of `model` implementing a method `reset_parameters()`. This 

16 is implemented for core `torch.nn` layers, but this may not be the case for custom 

17 implementations of exotic layers. A warning is logged for modules that do not implement 

18 `reset_parameters()`. 

19 

20 The actual logic of renitializing parameters depends on the type of layer. It may affect the 

21 module's buffers (non-trainable parameters, e.g., batch norm stats) as well. 

22 

23 Args: 

24 model: The model to be re-initialized. 

25 """ 

26 for module in model.modules(): 

27 # Skip modules without any parameters of their own. 

28 if not list(module.parameters(recurse=False)) and not list(module.buffers(recurse=False)): 

29 continue 

30 try: 

31 module.reset_parameters() 

32 except AttributeError: 

33 logging.warning(f"Skipping module {module} while resetting parameters.")

36def get_generator(seed: Optional[int] = None) -> torch.Generator: 

37 """Provides a torch.Generator for the given seed. 

38 

39 torch.default_generator is returned if seed is None. 

40 """ 

41 if seed is None: 

42 return torch.default_generator 

43 rng = torch.Generator(device="cpu") 

44 rng.manual_seed(seed) 

45 return rng

48def randomly_split_data( 

49 dataset: Dataset, proportions: List[float], seed: int = defaults.SEED 

50) -> List[Dataset]: 

51 """Randomly splits a dataset into chunks.""" 

52 rng = get_generator(seed) 

53 split_sizes = _proportions_into_sizes(proportions, len(dataset)) 

54 return random_split(dataset, split_sizes, generator=rng)

57def _proportions_into_sizes(proportions: List[float], size: int) -> List[int]: 

58 """A helper function to convert chunk proportions into sizes. 

59 

60 In case of rounding doubts, any remaining samples are appended to the last split size. 

61 """ 

62 assert sum(proportions) == 1.0 

63 sizes = [round(proportion * size) for proportion in proportions[:-1]] 

64 sizes.append(size - sum(sizes)) 

65 return sizes
